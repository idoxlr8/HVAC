function updatetax() {
	if (document.getElementById('taxable').checked == true)
		{
		document.getElementById('tax').value = formatCurrency((document.getElementById('labortotal').value -0) *.0825);
		alert ('Labor Tax Has Been Included In This Invoice.');
		}
		
	else if (document.getElementById('partstax').checked == true)
		{
		document.getElementById('tax').value = formatCurrency((document.getElementById('partstotal').value -0) *.0825);
		alert ('Parts Tax Has Been Included In This Invoice.');
		}
		
	else if (document.getElementById('taxable').checked == true && document.getElementById('partstax').checked == true)
		{
		document.getElementById('tax').value = formatCurrency((document.getElementById('subtotal').value -0) *.0825);
		alert ('Labor And Parts Tax Has Been Included In This Invoice.');
		}

	else
		{
		document.getElementById('tax').value = "";
		alert ('No Taxes Have Been Included In This Invoice.');
		}
}
